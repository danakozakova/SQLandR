Škálovanie údajov

Škálujte data o rozlohe a počtu obcí. Škálované dáta vložte do tabulky okres.

minRozloha <- min(tableOkres$ok_rozloha)
maxRozloha <- max(tableOkres$ok_rozloha) 
tableOkres$ok_rozloha_new <- (tableOkres$ok_rozloha-minRozloha)/(maxRozloha - minRozloha) 
minPocet <- min(tableOkres$ok_pocet_obci)
maxPocet <- max(tableOkres$ok_pocet_obci) 
tableOkres$ok_pocet_new <- (tableOkres$ok_pocet_obci-minPocet)/(maxPocet - minPocet)
tableOkres |> head(2)

plot(tableOkres$ok_rozloha, tableOkres$ok_pocet_obci,

text(tableOkres$ok_rozloha, tableOkres$ok_pocet_obci, label = tableOkres$ok_nazev_kratky)
